# https://github.com/DOMjudge/checktestdata/blob/main/doc/format-spec.md

INT(1, 1, t) NEWLINE
REP(t)
    NEWLINE
    INT(5, 5, r) SPACE
    INT(6, 6, c) NEWLINE

    INT(3, 3, n) NEWLINE

    REP(n)
        REGEX("[frht]") SPACE
        INT(0, r-1) SPACE
        INT(0, c-1) SPACE
        REGEX("[nsew]") NEWLINE
    END

    INT(1, 2000) SPACE
    INT(1, 200) NEWLINE
END
